Beskrivning

Implementera paginering för stora dataset för att förbättra laddningstider och användarupplevelse.

Implementation

# dashboard/backend/paginated_api.py
from flask import request

@app.route('/test-cases')
def get_test_cases():
page = int(request.args.get('page', 1))
per_page = int(request.args.get('per_page', 20))

# Query with pagination
test_cases = TestCase.query.paginate(page=page, per_page=per_page)

return {
'data': [tc.to_dict() for tc in test_cases.items],
'pagination': {
'page': page,
'per_page': per_page,
'total': test_cases.total,
'pages': test_cases.pages
}
}

# src/error_handlers.py
from flask import jsonify

class APIError(Exception):
def __init__(self, message, status_code=400, payload=None):
self.message = message
self.status_code = status_code
self.payload = payload

@app.errorhandler(APIError)
def handle_api_error(error):
response = {
'error': {
'code': error.status_code,
'message': error.message,
'details': error.payload
}
}
return jsonify(response), error.status_code

# Usage
raise APIError('Invalid input', 400, {'field': 'email'})

Fördelar

  • Konsekventa felmeddelanden
  • Bättre klienthantering
  • Enklare debugging

Implementation

# src/encryption.py
from cryptography.fernet import Fernet

key = Fernet.generate_key()
cipher = Fernet(key)

def encrypt_data(data):
return cipher.encrypt(data.encode()).decode()

def decrypt_data(encrypted_data):
return cipher.decrypt(encrypted_data.encode()).decode()

# Usage in models
class User(db.Model):
email = db.Column(db.String, nullable=False)

@property
def encrypted_email(self):
return encrypt_data(self.email)

@encrypted_email.setter
def encrypted_email(self, value):
self.email = decrypt_data(value)

Fördelar

  • Skydd av känslig data
  • Compliance med dataskyddsregler
  • Möjlighet att hantera krypterade backups
