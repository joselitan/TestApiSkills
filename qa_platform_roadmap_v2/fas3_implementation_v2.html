import graphene
from graphene import ObjectType, String, List

class TestCaseType(ObjectType):
id = String()
title = String()
status = String()
priority = String()

class Query(ObjectType):
test_cases = List(TestCaseType, priority=String())

def resolve_test_cases(self, info, priority=None):
query = TestCase.query
if priority:
query = query.filter_by(priority=priority)
return query.all()

# Flask route
@app.route('/graphql', methods=['POST'])

🏢 3.3 Multi-tenancy

Beskrivning

Isolera data per organisation/kund för SaaS-användning.

Implementation

# src/multi_tenant.py
from flask import g

@app.before_request
def set_tenant():
# Extract tenant from header or subdomain
tenant_id = request.headers.get('X-Tenant-ID')
g.tenant_id = tenant_id

class TenantMixin:
tenant_id = db.Column(db.String(50), nullable=False, default=lambda: g.tenant_id)

class TestCase(db.Model, TenantMixin):
title = db.Column(db.String(200))

@classmethod
def query_tenant(cls):
return cls.query.filter_by(tenant_id=g.tenant_id)

# Usage
@app.route('/test-cases')
def get_test_cases():
return TestCase.query_tenant().all()

from prometheus_client import Counter, Histogram, generate_latest

REQUEST_COUNT = Counter('http_requests_total', 'Total HTTP requests', ['method', 'endpoint'])
REQUEST_LATENCY = Histogram('http_request_duration_seconds', 'Request latency')

g.start_time = time.time()

@app.after_request
def after_request(response):
REQUEST_COUNT.labels(request.method, request.path).inc()
REQUEST_LATENCY.observe(time.time() - g.start_time)
return response
